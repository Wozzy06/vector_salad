def to_path
  Path.new(
    N.n(@x, @y),
    N.n(@x, @y + @height),
    N.n(@x + @width, @y + @height),
    N.n(@x + @width, @y),
    **@options
  )
end

def to_svg
  svg = "<rect x=\"#{at[0]}\" y=\"#{at[1]}\""
  svg << " width=\"#{width}\" height=\"#{height}\""
  svg << VectorSalad::Exporters::SvgExporter.options(@options)
  svg << "/>"
end
