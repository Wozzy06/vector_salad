def initialize(width = nil, height, **options)
  width = height * 2 if width.nil?
  @width, @height = width, height
  @options = options
  @x, @y = 0, 0
  self
end

def to_path
  Path.new(
    N.n(@x, @y),
    N.n(@x - @width / 2, @y + @height),
    N.n(@x + @width / 2, @y + @height),
    **@options
  )
end
