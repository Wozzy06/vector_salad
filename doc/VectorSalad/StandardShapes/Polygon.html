def to_path
  nodes = []
  angle = 2 * Math::PI / @sides
  theta = angle / 2 + Math::PI / 2
  @sides.times do |n|
    nodes[n] = []
    nodes[n][0] = @radius * Math.cos(angle * n + theta) + @x
    nodes[n][1] = @radius * Math.sin(angle * n + theta) + @y
  end

  Path.new(*nodes, **@options)
end
